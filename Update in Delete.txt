        [HttpPut("editPerson")]
        [Consumes("application/json")]
        public async Task<ActionResult<Oseba>> Put(Oseba oseba)
        {
            try
            {
                _context.Update(oseba);
                await _context.SaveChangesAsync();
                var posodobljenaOseba = await _context.Osebe.FirstOrDefaultAsync(m => m.Id == oseba.Id);
                return Ok(posodobljenaOseba);
            }
            catch (Exception ex)
            {
                _logger.LogError($"Edit failed: \n{ex}");
                return StatusCode(StatusCodes.Status500InternalServerError, "Internal Server Error");
            }  
        }

        [HttpDelete("deletePerson")]
        [Consumes("application/json")]
        public async Task<ActionResult<Oseba>> Delete(Oseba oseba)
        {
            try
            {
                _context.Osebe.Remove(oseba);
                await _context.SaveChangesAsync();
                return Ok(oseba);
            }
            catch (Exception ex)
            {
                _logger.LogError($"Edit failed: \n{ex}");
                return StatusCode(StatusCodes.Status500InternalServerError, "Internal Server Error");
            }
        }
